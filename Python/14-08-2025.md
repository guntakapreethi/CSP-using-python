6) Check if a number is prime or not.
ANS) num = int(input("Enter a number: "))
if num <= 1:
    print("Not a prime number")
else:
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            print("Not a prime number")
            break
    else:
        print("Prime number")


7) Find the factorial of a number.
ANS) num = int(input("Enter a number: "))
factorial = 1
if num < 0:
    print("Factorial doesn't exist negative numbers")
elif num == 0:
    print("The factorial of 0 is 1")
else:
    for i in range(1, num + 1):
        factorial *= i
    print("The factorial of", num, "is", factorial)



8) Print the Fibonacci series upto n terms.
ANS) n = int(input("Enter the number of terms: "))
a, b = 0, 1
count = 0
if n <= 0:
    print("Please enter a positive integer")
else:
    print("Fibonacci series:")
    while count < n:
        print(a, end=" ")
        a, b = b, a + b
        count += 1


9) Find the sum of digits of a given number
ANS) num = int(input("Enter a number: "))
sum_of_digits = 0
while num > 0:
    digit = num % 10
    sum_of_digits += digit
    num = num // 10
print("Sum of digits is:", sum_of_digits)


10) Reverse the digits of a given number.
ANS) num = int(input("Enter a number: "))
reverse = 0
while num > 0:
    digit = num % 10
    reverse = reverse * 10 + digit
    num = num // 10
print("Reversed number is:", reverse)
